% ---------- Hybrid autoscale wrapper ----------
% Usage: \begin{FitOnePage} ... entire resume ... \end{FitOnePage}
\newenvironment{FitOnePage}
{%
  % Put all content into \resumecontent box first
  \newsavebox{\resumecontent}%
  \begin{lrbox}{\resumecontent}%
  \begin{minipage}{\textwidth}%
}
{%
  \end{minipage}%
  \end{lrbox}%
  % Now measure and scale *only if* too tall.
  \ifdim\ht\resumecontent>\textheight
    % Scale uniformly down so total height == \textheight (width scales same factor)
    \begin{center}
      \adjustbox{max totalsize={\textwidth}{\textheight},center}{\usebox{\resumecontent}}
    \end{center}
  \else
    \usebox{\resumecontent}%
  \fi
}
